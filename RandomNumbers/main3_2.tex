\section{Problem Statement}
\begin{frame}{Problem Statement}
    \begin{block}{(2.5)} Find a theoretical expression for $F_V(x)$, where $V = -2\ln(1-U)$.
    \end{block}
\end{frame}

\section{Solution}
\begin{frame}{Solution}
We have:

\begin{align}
F_V(x) &= \pr{V \leq x} \\
&= \pr{-2\ln(1-U) \leq x} \\
&= \pr{1-U \geq	\exp{\left(-\frac{x}{2}\right)}} \\
&= \pr{U \leq 1 - \exp{\left(-\frac{x}{2}\right)}} \\
&= F_U\left(1 - \exp{\left(-\frac{x}{2}\right)}\right) 
\end{align}

\end{frame}



\begin{frame}{Solution}
Therefore,
\begin{align}
   F_V(x) =
    \begin{cases}
        0, & 1 - \exp{\left(-\frac{x}{2}\right)} \in (-\infty,0) \\
        1 - \exp{\left(-\frac{x}{2}\right)}, & 1 - \exp{\left(-\frac{x}{2}\right)} \in (0,1) \\
        1, & 1 - \exp{\left(-\frac{x}{2}\right)} \in (1, \infty)
    \end{cases}
\end{align}

From this we get:

\begin{align}
   F_V(x) =
    \begin{cases}
        0, & x \in (-\infty,0) \\
        1 - \exp{\left(-\frac{x}{2}\right)}, & x \in (0,\infty) 
    \end{cases}
\end{align}
\end{frame}
